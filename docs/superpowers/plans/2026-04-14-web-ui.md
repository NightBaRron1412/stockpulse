> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a premium Dark Glass web dashboard for StockPulse with FastAPI backend and Next.js frontend — 7 pages covering dashboard, watchlist, portfolio, signals, validation, reports, and settings.

**Architecture:** FastAPI wraps existing StockPulse Python modules as a REST API. Next.js 15 App Router frontend consumes the API. No database — reads directly from existing YAML configs, JSON state files, and scan outputs. Polling for updates (30s dashboard, 5s during scans).

**Tech Stack:** FastAPI, uvicorn, Next.js 15, TypeScript, Tailwind CSS, shadcn/ui, Recharts, react-markdown, remark-gfm

"""FastAPI server wrapping StockPulse modules."""
import json
import logging
from datetime import datetime
from pathlib import Path
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI(title="StockPulse API", version="0.1.0")
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])

logger = logging.getLogger(__name__)
PROJECT_ROOT = Path(__file__).resolve().parent.parent.parent

# --- Dashboard ---
@app.get("/api/dashboard")
def get_dashboard():
    from stockpulse.portfolio.tracker import get_portfolio_status
    from stockpulse.config.settings import load_watchlists
    portfolio = get_portfolio_status()
    # Load latest scan results from JSON output
    json_dir = PROJECT_ROOT / "outputs" / "json"
    today = datetime.now().strftime("%Y-%m-%d")
    scan_file = json_dir / f"{today}-morning.json"
    top_signals = []
    if scan_file.exists():
        with open(scan_file) as f:
            data = json.load(f)
        top_signals = data.get("recommendations", [])[:10]
    activity = _get_activity()
    scan_status = _get_scan_status()
    # Count by action
    all_recs = []
    if scan_file.exists():
        with open(scan_file) as f:
            all_recs = json.load(f).get("recommendations", [])
    counts = {}
    for r in all_recs:
        a = r.get("action", "HOLD")
        counts[a] = counts.get(a, 0) + 1
    return {"portfolio": portfolio, "top_signals": top_signals, "activity": activity,
            "scan_status": scan_status, "signal_count": counts}

# --- Watchlist ---
@app.get("/api/watchlist")
def get_watchlist():
    from stockpulse.config.settings import load_watchlists
    wl = load_watchlists()
    # Load latest scan for scores
    json_dir = PROJECT_ROOT / "outputs" / "json"
    today = datetime.now().strftime("%Y-%m-%d")
    scan_file = json_dir / f"{today}-morning.json"
    recs = {}
    if scan_file.exists():
        with open(scan_file) as f:
            for r in json.load(f).get("recommendations", []):
                recs[r["ticker"]] = r
    result = []
    for ticker in wl.get("user", []):
        rec = recs.get(ticker, {"ticker": ticker, "action": "UNKNOWN", "composite_score": 0, "confidence": 0})
        rec["source"] = "user"
        result.append(rec)
    for ticker in wl.get("discovered", []):
        if ticker not in {r["ticker"] for r in result}:
            rec = recs.get(ticker, {"ticker": ticker, "action": "UNKNOWN", "composite_score": 0, "confidence": 0})
            rec["source"] = "discovered"
            result.append(rec)
    result.sort(key=lambda x: x.get("composite_score", 0), reverse=True)
    return result

@app.get("/api/watchlist/{ticker}")
def get_watchlist_ticker(ticker: str):

    df = get_price_history(ticker, period="1y")

@app.get("/api/reports")
def list_reports():
    reports_dir = PROJECT_ROOT / "outputs" / "reports"
    if not reports_dir.exists():
        return []
    result = []
    for f in sorted(reports_dir.glob("*.md"), reverse=True):
        name = f.stem
        parts = name.split("-")
        date = "-".join(parts[:3]) if len(parts) >= 3 else name
        rtype = parts[3] if len(parts) > 3 else "unknown"
        result.append({"filename": f.name, "date": date, "type": rtype, "title": f.name})
    return result

@app.get("/api/alerts/recent")
def get_recent_alerts():
    log_path = PROJECT_ROOT / "outputs" / "logs" / "alerts.log"
    if not log_path.exists():
        return []
    lines = log_path.read_text().strip().split("\n")
    alerts = []
    for line in lines[-50:]:
        try:
            alerts.append(json.loads(line))
        except Exception:
            pass
    alerts.reverse()
    return alerts

@app.get("/api/activity")
def get_activity():
    return _get_activity()

def _get_activity():
    log_path = PROJECT_ROOT / "outputs" / "logs" / "stockpulse.log"
    if not log_path.exists():
        return []
    lines = log_path.read_text().strip().split("\n")
    events = []
    for line in lines[-200:]:
        if any(kw in line for kw in ["MORNING SCAN", "Intraday", "EOD", "SEC", "Portfolio",
                                      "discovered", "changes detected", "complete", "milestone"]):
            parts = line.split(" ", 2)
            events.append({"timestamp": parts[0] if parts else "", "type": "system",
                           "message": parts[2] if len(parts) > 2 else line})
    return events[-50:]

# --- Scan ---
@app.get("/api/scan/status")
def get_scan_status():
    return _get_scan_status()

def _get_scan_status():
    log_path = PROJECT_ROOT / "outputs" / "logs" / "stockpulse.log"
    last_completed = "Never"
    running = False
    progress = ""
    if log_path.exists():
        lines = log_path.read_text().strip().split("\n")
        for line in reversed(lines[-100:]):
            if "Scan complete" in line and not last_completed != "Never":
                last_completed = line.split(" ")[0]
                break
            if "MORNING SCAN START" in line:
                running = True
            if "Scanned " in line and "/" in line:
                progress = line.split("Scanned ")[1].split(" ")[0] if "Scanned " in line else ""
                running = True
    return {"running": running, "progress": progress, "last_completed": last_completed, "next_scheduled": "09:35 ET"}

@app.post("/api/scan")
def trigger_scan():
    import threading
    def _run():
        from stockpulse.scheduler.jobs import morning_scan_job
        morning_scan_job()
    threading.Thread(target=_run, daemon=True).start()
    return {"status": "started"}

# --- Quote and History ---
@app.get("/api/quote/{ticker}")
def get_quote(ticker: str):
    from stockpulse.data.provider import get_current_quote
    return get_current_quote(ticker)

@app.get("/api/history/{ticker}")
def get_history(ticker: str, period: str = "6mo"):
    from stockpulse.data.provider import get_price_history
    df = get_price_history(ticker, period=period)
    if df.empty:
        raise HTTPException(404, f"No data for {ticker}")
    return {"dates": [d.isoformat() for d in df.index], "close": df["Close"].tolist(),
            "volume": df["Volume"].tolist(), "high": df["High"].tolist(), "low": df["Low"].tolist()}

# --- Config ---
@app.get("/api/config")
def get_config():
    from stockpulse.config.settings import load_strategies
    return load_strategies()

# --- Watchlist management ---
@app.post("/api/watchlist/add")
def add_to_watchlist(data: dict):
    from stockpulse.config.settings import load_watchlists, save_watchlists
    ticker = data.get("ticker", "").upper()
    if not ticker:
        raise HTTPException(400, "ticker required")
    wl = load_watchlists()
    if ticker not in wl.get("user", []):
        wl.setdefault("user", []).append(ticker)
        save_watchlists(wl)
    return {"status": "added", "ticker": ticker}

@app.post("/api/watchlist/remove")
def remove_from_watchlist(data: dict):
    from stockpulse.config.settings import load_watchlists, save_watchlists
    ticker = data.get("ticker", "").upper()
    wl = load_watchlists()
    if ticker in wl.get("user", []):
        wl["user"].remove(ticker)
    if ticker in wl.get("discovered", []):
        wl["discovered"].remove(ticker)
    wl["priority"] = [p for p in wl.get("priority", []) if p.get("ticker") != ticker]
    save_watchlists(wl)
    return {"status": "removed", "ticker": ticker}

```typescript
import { useState, useEffect, useCallback } from "react";

export function usePolling<T>(fetcher: () => Promise<T>, intervalMs: number = 30000) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refresh = useCallback(async () => {
    try {
      const result = await fetcher();
      setData(result);
      setError(null);
    } catch (e: any) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, [fetcher]);

  useEffect(() => {
    refresh();
    const id = setInterval(refresh, intervalMs);
    return () => clearInterval(id);
  }, [refresh, intervalMs]);

  return { data, loading, error, refresh };
}
```
